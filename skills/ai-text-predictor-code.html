AI Text Predictor - Source Code

<input type="text" id="textInput" placeholder="Start typing..." />

<script>
const wordsJSON = {
  "A": ["Apple", "Ant", "Artist", "Arcade"],
  "B": ["Ball", "Book", "Bridge", "Bread"],
  "C": ["Code", "Canvas", "Click", "Client"]
};

const textInput = document.getElementById("textInput");
const suggestionsBox = document.getElementById("suggestions");
let allWords = [];

Object.values(wordsJSON).forEach(arr =>
  allWords.push(...arr.map(w => w.toLowerCase()))
);

textInput.addEventListener("input", () => {
  const value = textInput.value.toLowerCase();
  suggestionsBox.innerHTML = "";
  if (!value) return;

  const matches = allWords.filter(word => word.startsWith(value)).slice(0, 5);
  matches.forEach(match => {
    const div = document.createElement("div");
    div.className = "suggestion";
    div.textContent = match;
    div.onclick = () => {
      textInput.value = match;
      suggestionsBox.innerHTML = "";
    };
    suggestionsBox.appendChild(div);
  });
});
</script>
